let _histData=null;
let _currentFilter='all';
let _agentsMap={};     // agent_id -> agent config (incl. is_primary, display_name)
let _primaryAgentId=null;

async function loadAgentsMap(){
  try{
    const r=await fetch('/api/agents');
    if(!r.ok)return;
    const agents=await r.json();
    _agentsMap={};
    (agents||[]).forEach(a=>{
      _agentsMap[a.agent_id]=a;
      if(a.is_primary)_primaryAgentId=a.agent_id;
    });
  }catch(e){console.error('loadAgentsMap',e)}
}

async function loadHistory(){
  try{
    await loadAgentsMap();
    const r=await fetch('/api/stats_history?limit=200');
    if(!r.ok)return;
    _histData=await r.json();
    if(_histData.error)return;
    renderStats(_histData.summary);
    renderHistTable();
  }catch(e){console.error('loadHistory',e)}
}

  pEl.style.color=s.total_pnl>=0?'var(--g)':'var(--r)';

  wrEl.textContent=s.real_win_rate+'%';
  wrEl.style.color=s.real_win_rate>=55?'var(--g)':s.real_win_rate>=45?'var(--y)':'var(--r)';

function renderAgentBreakdown(){
  if(!_histData||!_histData.table)return;
  const byAgent={};
  _histData.table.forEach(r=>{
    const m=r.ai_model||_primaryAgentId||'primary';
    if(!byAgent[m])byAgent[m]={wins:0,losses:0,pnl:0,count:0};
    byAgent[m].count++;
    if(r.won)byAgent[m].wins++;else byAgent[m].losses++;
    if(!r.is_ghost&&!r.is_shadow)byAgent[m].pnl+=(r.profit||0);
  });
  const el=$('agent-breakdown');
  if(!el)return;
  let html='';
  for(const[agent,s] of Object.entries(byAgent)){
    const wr=s.count>0?Math.round(s.wins/s.count*100):0;
    const color=wr>=55?'var(--g)':wr>=45?'var(--y)':'var(--r)';
    const pColor=s.pnl>=0?'var(--g)':'var(--r)';
    const cfg=_agentsMap[agent]||{};
    const label=(cfg.display_name||agent.replace(/_/g,' ')).toUpperCase();
    const primaryTag=cfg.is_primary?' • PRIMARY':'';
    html+=`

      ${label}${primaryTag}
      ${s.count} trades
      ${wr}% WR
      ${s.pnl>=0?'+':''}$${fm(Math.abs(s.pnl))}
      (${s.wins}W/${s.losses}L)
    
`;
  }
  el.innerHTML=html||'Sin datos';
}

// Deterministic color assigned by agent api_type (anthropic=cyan, openai=green, local=g, other=b2 primary-like)
function _colorForAgent(agentId){
  const cfg=_agentsMap[agentId];
  if(!cfg)return 'var(--b2)';
  if(cfg.is_primary)return 'var(--b2)';        // primary = institutional blue
  const t=(cfg.api_type||'').toLowerCase();
  if(t==='anthropic')return 'var(--cyan)';
  if(t==='openai')return 'var(--y)';
  if(t==='local')return 'var(--g)';
  return 'var(--b2)';
}

function setFilter(f,btn){

}

function utcToLocal(utcStr){
  if(!utcStr||utcStr.length<16)return '--';
  // created_at is stored as 'YYYY-MM-DD HH:MM:SS' in UTC
  const d=new Date(utcStr+'Z');
  if(isNaN(d))return utcStr.substring(11,16);
  return d.toLocaleTimeString([],{hour:'2-digit',minute:'2-digit',hour12:false});
}

async function loadWindowTrace(){
  const wid=parseInt($('wt-id').value);
  const st=$('wt-status');const sum=$('wt-summary');const ag=$('wt-agents');const raw=$('wt-raw');
  if(!wid){st.textContent='Falta window_id';st.style.color='var(--r)';return}
  st.textContent='Cargando...';st.style.color='var(--t3)';
  sum.innerHTML='';ag.innerHTML='';raw.textContent='';
  try{
    const r=await fetch('/api/window_trace/'+wid);const d=await r.json();
    if(!d.ok){st.textContent=d.error||'Error';st.style.color='var(--r)';return}
    st.textContent='OK';st.style.color='var(--g)';
    const w=d.window||{};const c=d.counts||{};
    sum.innerHTML=
      'Window '+wid+' — open=$'+(w.open_price||0).toFixed(2)+
      ' close=$'+(w.close_price||0).toFixed(2)+
      ' outcome='+(w.outcome||'?')+''+
      ' | preds='+c.predictions+' bets='+c.bets+' real='+c.real_bets+' shadow='+c.shadow_bets+' ghost='+c.ghost_bets+' liq='+c.liquidations;
    let html='';
    (d.agents||[]).forEach(a=>{
      const dir=a.prediction||'?';const ok=a.correct;
      const dirColor=dir==='UP'?'var(--g)':dir==='DOWN'?'var(--r)':'var(--t3)';
      const okBadge=ok===1?'W':ok===0?'L':'';
      html+='
';
      html+='
';
      html+=''+a.agent+'';
      html+=''+dir+' '+(a.confidence||0)+'%';
      if(a.strategy)html+=''+a.strategy+'';
      html+=okBadge;
      if(a.bets&&a.bets.length){a.bets.forEach(b=>{
        const p=(b.profit||0);
        html+=''+
              b.side+' $'+(b.amount||0).toFixed(2)+' @'+(b.odds_cents||0).toFixed(0)+'c '+
              (p>=0?'+':'')+'$'+p.toFixed(2)+'';
      })}
      html+='
';
      if(a.reasoning)html+='
'+escapeHtml(a.reasoning)+'
';
      html+='
';
    });
    ag.innerHTML=html||'
Sin agentes
';
    raw.textContent=JSON.stringify(d,null,2);
  }catch(e){st.textContent='Error red';st.style.color='var(--r)'}
}
